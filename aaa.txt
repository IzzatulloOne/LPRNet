def find_first_existing(*relative_candidates: str) -> Path:
    for base in ROOT_CANDIDATES:
        for rel in relative_candidates:
            candidate = base / rel
            if candidate.exists():
                return candidate
    raise FileNotFoundError(
        "None of the candidate paths exists:\n"
        + "\n".join(str(Path(base) / rel) for base in ROOT_CANDIDATES for rel in relative_candidates)
    )

def iter_valid_labels(folder: Path, allowed_chars):
    allowed = set(allowed_chars)
    for fpath in sorted(folder.glob("*.jpg")):
        label = fpath.stem.split("_")[0].upper()
        if len(label) != EXPECTED_LEN:
            continue
        if any(char not in allowed for char in label):
            continue
        yield fpath, label


def infer_chars(train_dir: Path, val_dir: Path):
    if not AUTO_CHARS_FROM_DATA:
        return DEFAULT_CHARS

    discovered = set()
    for folder in [train_dir, val_dir]:
        for fpath in sorted(folder.glob("*.jpg")):
            label = fpath.stem.split("_")[0].upper()
            if len(label) != EXPECTED_LEN:
                continue
            discovered.update(label)

    chars = sorted(discovered, key=lambda value: DEFAULT_CHARS.index(value))
    return chars if chars else DEFAULT_CHARS


CHARS = infer_chars(TRAIN_DIR, VAL_DIR)
BLANK_IDX = 0
CHARS_DICT = {char: index + 1 for index, char in enumerate(CHARS)}
IDX_TO_CHAR = {index + 1: char for index, char in enumerate(CHARS)}
NUM_CLASSES = len(CHARS) + 1


def scan_folder(folder: Path):
    ok = 0
    wrong_len = []
    bad = []

    for fpath in sorted(folder.glob("*.jpg")):
        label = fpath.stem.split("_")[0].upper()
        if len(label) != EXPECTED_LEN:
            wrong_len.append((fpath.name, label, len(label)))
            continue

        bad_chars = [char for char in label if char not in CHARS_DICT]
        if bad_chars:
            bad.append((fpath.name, label, bad_chars))
            continue

        ok += 1

    return ok, wrong_len, bad


def motion_blur(image: np.ndarray, ksize: int) -> np.ndarray:
    kernel = np.zeros((ksize, ksize), dtype=np.float32)
    if random.random() < 0.5:
        kernel[ksize // 2, :] = 1.0
    else:
        kernel[:, ksize // 2] = 1.0
    kernel /= kernel.sum()
    return cv2.filter2D(image, -1, kernel)


def jpeg_artifacts(image: np.ndarray, quality: int) -> np.ndarray:
    ok, encoded = cv2.imencode(".jpg", image, [int(cv2.IMWRITE_JPEG_QUALITY), quality])
    if not ok:
        return image
    decoded = cv2.imdecode(encoded, cv2.IMREAD_COLOR)
    return decoded if decoded is not None else image


def mild_perspective(image: np.ndarray) -> np.ndarray:
    height, width = image.shape[:2]
    shift_x = width * random.uniform(0.01, 0.05)
    shift_y = height * random.uniform(0.01, 0.08)

    src = np.float32([
        [0, 0],
        [width - 1, 0],
        [width - 1, height - 1],
        [0, height - 1],
    ])
    dst = np.float32([
        [random.uniform(0, shift_x), random.uniform(0, shift_y)],
        [width - 1 - random.uniform(0, shift_x), random.uniform(0, shift_y)],
        [width - 1 - random.uniform(0, shift_x), height - 1 - random.uniform(0, shift_y)],
        [random.uniform(0, shift_x), height - 1 - random.uniform(0, shift_y)],
    ])
    matrix = cv2.getPerspectiveTransform(src, dst)
    return cv2.warpPerspective(
        image,
        matrix,
        (width, height),
        flags=cv2.INTER_LINEAR,
        borderMode=cv2.BORDER_REPLICATE,
    )


def augment_image(image: np.ndarray) -> np.ndarray:
    augmented = image.copy()

    if random.random() < 0.45:
        alpha = random.uniform(0.8, 1.2)
        beta = random.uniform(-18, 18)
        augmented = cv2.convertScaleAbs(augmented, alpha=alpha, beta=beta)

    if random.random() < 0.20:
        gamma = random.uniform(0.8, 1.25)
        table = np.array([((value / 255.0) ** gamma) * 255 for value in range(256)], dtype=np.uint8)
        augmented = cv2.LUT(augmented, table)

    if random.random() < 0.18:
        augmented = mild_perspective(augmented)

    if random.random() < 0.18:
        augmented = cv2.GaussianBlur(augmented, (3, 3), 0)

    if random.random() < 0.12:
        augmented = motion_blur(augmented, ksize=3)

    if random.random() < 0.18:
        noise = np.random.normal(0, 6, augmented.shape).astype(np.float32)
        augmented = np.clip(augmented.astype(np.float32) + noise, 0, 255).astype(np.uint8)

    if random.random() < 0.20:
        quality = random.randint(45, 80)
        augmented = jpeg_artifacts(augmented, quality=quality)

    return augmented


class UZPlateDataset(Dataset):
    def __init__(self, data_dir: Path, augment: bool):
        self.samples = []
        self.augment = augment

        for fpath, label in iter_valid_labels(data_dir, CHARS):
            self.samples.append((str(fpath), label))

        if not self.samples:
            raise RuntimeError(f"No valid samples found in: {data_dir}")

    def __len__(self):
        return len(self.samples)

    def __getitem__(self, index):
        fpath, label = self.samples[index]

        with open(fpath, "rb") as file:
            buffer = np.frombuffer(file.read(), dtype=np.uint8)

        image = cv2.imdecode(buffer, cv2.IMREAD_COLOR)
        if image is None:
            raise RuntimeError(f"Failed to decode image: {fpath}")

        image = cv2.resize(image, IMG_SIZE)

        if self.augment:
            image = augment_image(image)

        image = image.astype(np.float32)
        image -= 127.5
        image *= 0.0078125

        tensor = torch.from_numpy(image).permute(2, 0, 1)
        return tensor, label, fpath


def collate_fn(batch):
    images, labels, paths = zip(*batch)
    return torch.stack(images), list(labels), list(paths)


def encode(labels):
    targets = []
    lengths = []

    for label in labels:
        encoded = [CHARS_DICT[char] for char in label]
        targets.extend(encoded)
        lengths.append(len(encoded))

    return (
        torch.tensor(targets, dtype=torch.long, device=DEVICE),
        torch.tensor(lengths, dtype=torch.long, device=DEVICE),
    )


def decode(logits):
    predictions = logits.argmax(dim=1).detach().cpu().numpy()
    results = []

    for prediction in predictions:
        chars = []
        prev = -1
        for value in prediction:
            value = int(value)
            if value != prev and value != BLANK_IDX:
                chars.append(IDX_TO_CHAR.get(value, ""))
            prev = value
        results.append("".join(chars))

    return results


def exact_match_acc(logits, labels):
    predicted_texts = decode(logits)
    correct = sum(pred == target for pred, target in zip(predicted_texts, labels))
    return correct, len(labels), predicted_texts


def char_accuracy(predicted_texts, labels):
    correct = 0
    total = 0
    for prediction, target in zip(predicted_texts, labels):
        for pred_char, target_char in zip(prediction, target):
            if pred_char == target_char:
                correct += 1
        total += len(target)
    return correct, total


def build_position_stats(samples):
    per_position_counts = [Counter() for _ in range(EXPECTED_LEN)]
    per_char_counts = Counter()

    for _path, label in samples:
        for position, char in enumerate(label):
            per_position_counts[position][char] += 1
            per_char_counts[char] += 1

    return per_position_counts, per_char_counts


def build_balanced_sampler(dataset):
    per_position_counts, _ = build_position_stats(dataset.samples)
    weights = []

    for _path, label in dataset.samples:
        score = 0.0
        for position, char in enumerate(label):
            freq = per_position_counts[position][char]
            score += 1.0 / math.sqrt(max(freq, 1))
        weights.append(score / EXPECTED_LEN)

    weights_tensor = torch.tensor(weights, dtype=torch.double)
    return WeightedRandomSampler(weights_tensor, num_samples=len(weights_tensor), replacement=True)


def position_charset_string(samples):
    per_position_counts, per_char_counts = build_position_stats(samples)
    position_sets = []
    for position, counts in enumerate(per_position_counts):
        chars = "".join(sorted(counts.keys(), key=lambda value: DEFAULT_CHARS.index(value)))
        position_sets.append(f"{position}:{chars}")
    present = "".join(sorted(per_char_counts.keys(), key=lambda value: DEFAULT_CHARS.index(value)))
    return present, position_sets


def split_trainable_params(model):
    head_keywords = ("container", "classifier", "head", "fc")
    backbone_params = []
    head_params = []

    for name, parameter in model.named_parameters():
        if any(keyword in name.lower() for keyword in head_keywords):
            head_params.append(parameter)
        else:
            backbone_params.append(parameter)

    if not head_params:
        return [{"params": model.parameters(), "lr": LR}]

    return [
        {"params": backbone_params, "lr": BACKBONE_LR},
        {"params": head_params, "lr": LR},
    ]


def set_head_only_trainable(model):
    head_keywords = ("container", "classifier", "head", "fc")
    found = False
    for name, parameter in model.named_parameters():
        is_head = any(keyword in name.lower() for keyword in head_keywords)
        parameter.requires_grad = is_head
        found = found or is_head

    if not found:
        for parameter in model.parameters():
            parameter.requires_grad = True


def set_all_trainable(model):
    for parameter in model.parameters():
        parameter.requires_grad = True


def load_pretrained(model):
    if not USE_PRETRAINED or not PRETRAINED_PATH.exists():
        print("Training from scratch.")
        return

    try:
        state = torch.load(PRETRAINED_PATH, map_location=DEVICE, weights_only=True)
    except TypeError:
        state = torch.load(PRETRAINED_PATH, map_location=DEVICE)

    if not isinstance(state, dict):
        print("Pretrained file is not a state_dict-like object. Training from scratch.")
        return

    model_state = model.state_dict()
    loaded = {}
    skipped = []

    for key, value in state.items():
        if key in model_state and hasattr(value, "shape") and value.shape == model_state[key].shape:
            loaded[key] = value
        else:
            skipped.append(key)

    model_state.update(loaded)
    model.load_state_dict(model_state)
    print(f"Loaded layers : {len(loaded)}")
    print(f"Skipped layers: {len(skipped)}")


def build_optimizer(model):
    if USE_PRETRAINED:
        param_groups = split_trainable_params(model)
        return optim.AdamW(param_groups, weight_decay=WEIGHT_DECAY)
    return optim.AdamW(model.parameters(), lr=LR, weight_decay=WEIGHT_DECAY)


def should_replace_best(best_exact, best_char, val_exact, val_char):
    if val_exact > best_exact:
        return True
    if val_exact == best_exact and val_char > best_char:
        return True
    return False


def main():
    print(f"Device       : {DEVICE}")
    print(f"CUDA avail   : {torch.cuda.is_available()}")
    if torch.cuda.is_available():
        print(f"GPU          : {torch.cuda.get_device_name(0)}")
    print(f"Chars        : {''.join(CHARS)}")
    print(f"Classes      : {NUM_CLASSES} (blank=0)")
    print(f"Plate length : {EXPECTED_LEN}")
    print(f"Use augment  : {USE_AUGMENT}")
    print(f"Use sampler  : {USE_BALANCED_SAMPLER}")
    print(f"Use pretrained: {USE_PRETRAINED}")

    for split_name, folder in [("TRAIN", TRAIN_DIR), ("VAL", VAL_DIR)]:
        ok, wrong_len, bad = scan_folder(folder)
        print(f"\n=== {split_name} ===")
        print(f"OK labels: {ok}")
        if wrong_len:
            print(f"Wrong length: {len(wrong_len)}")
            for fname, label, length in wrong_len[:10]:
                print(f"  {fname} -> '{label}' (len={length})")
        if bad:
            print(f"Bad chars: {len(bad)}")
            for fname, label, chars in bad[:10]:
                print(f"  {fname} -> '{label}' bad={chars}")

    train_ds = UZPlateDataset(TRAIN_DIR, augment=USE_AUGMENT)
    val_ds = UZPlateDataset(VAL_DIR, augment=False)

    present_chars, position_sets = position_charset_string(train_ds.samples)
    print(f"\nTrain chars  : {present_chars}")
    print("Per-position :", " | ".join(position_sets))
    print(f"Train samples: {len(train_ds)}")
    print(f"Val samples  : {len(val_ds)}")

    train_sampler = build_balanced_sampler(train_ds) if USE_BALANCED_SAMPLER else None

    train_dl = DataLoader(
        train_ds,
        batch_size=BATCH_SIZE,
        shuffle=(train_sampler is None),
        sampler=train_sampler,
        collate_fn=collate_fn,
        num_workers=0,
        pin_memory=PIN_MEMORY,
        persistent_workers=False,
    )
    val_dl = DataLoader(
        val_ds,
        batch_size=BATCH_SIZE,
        shuffle=False,
        collate_fn=collate_fn,
        num_workers=0,
        pin_memory=PIN_MEMORY,
        persistent_workers=False,
    )

    model = build_lprnet(
        lpr_max_len=EXPECTED_LEN,
        phase=False,
        class_num=NUM_CLASSES,
        dropout_rate=DROPOUT_RATE,
    ).to(DEVICE)
    print("Model device :", next(model.parameters()).device)

    load_pretrained(model)

    if USE_PRETRAINED and FREEZE_BACKBONE_EPOCHS > 0:
        set_head_only_trainable(model)
        print(f"Warmup: head-only for first {FREEZE_BACKBONE_EPOCHS} epochs")
    else:
        set_all_trainable(model)

    ctc_loss = torch.nn.CTCLoss(blank=BLANK_IDX, reduction="mean", zero_infinity=True)
    optimizer = build_optimizer(model)
    scheduler = optim.lr_scheduler.ReduceLROnPlateau(
        optimizer,
        mode="max",
        factor=LR_FACTOR,
        patience=LR_PATIENCE,
        min_lr=MIN_LR,
    )
    scaler = torch.amp.GradScaler("cuda", enabled=(DEVICE.type == "cuda"))

    best_exact = -1.0
    best_char = -1.0
    epochs_without_improvement = 0
    shape_printed = False

    for epoch in range(1, EPOCHS + 1):
        if epoch == FREEZE_BACKBONE_EPOCHS + 1 and USE_PRETRAINED and FREEZE_BACKBONE_EPOCHS > 0:
            set_all_trainable(model)
            optimizer = build_optimizer(model)
            scheduler = optim.lr_scheduler.ReduceLROnPlateau(
                optimizer,
                mode="max",
                factor=LR_FACTOR,
                patience=LR_PATIENCE,
                min_lr=MIN_LR,
            )
            scaler = torch.amp.GradScaler("cuda", enabled=(DEVICE.type == "cuda"))
            print("Backbone unfrozen. Fine-tuning all layers now.")

        model.train()
        train_loss = 0.0

        for batch_index, (images, labels, paths) in enumerate(train_dl):
            images = images.to(DEVICE, non_blocking=True)
            optimizer.zero_grad(set_to_none=True)

            with torch.amp.autocast("cuda", enabled=(DEVICE.type == "cuda")):
                raw_logits = model(images)
                if not shape_printed:
                    print(f"\nDEBUG logits shape [B,C,T]: {tuple(raw_logits.shape)}")
                    print(f"DEBUG first labels          : {labels[:5]}")
                    print(f"DEBUG first paths           : {paths[:3]}")
                    shape_printed = True

                logits = raw_logits.permute(2, 0, 1)
                log_probs = torch.nn.functional.log_softmax(logits, dim=2).contiguous()
                targets, target_lengths = encode(labels)
                input_lengths = torch.full(
                    (images.size(0),),
                    logits.size(0),
                    dtype=torch.long,
                    device=DEVICE,
                )
                loss = ctc_loss(log_probs, targets, input_lengths, target_lengths)

            scaler.scale(loss).backward()
            scaler.unscale_(optimizer)
            torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            scaler.step(optimizer)
            scaler.update()

            train_loss += loss.item()

            if batch_index == 0 and epoch in {1, 2, 5, 10, 20, 40, 60, 80, 100, 120}:
                preview = decode(raw_logits[:5])
                print(f"\nTrain batch preview @ epoch {epoch}")
                print("GT:", labels[:5])
                print("PR:", preview)

        model.eval()
        exact_correct = 0
        exact_total = 0
        char_correct = 0
        char_total = 0
        sample_pred = None
        sample_gt = None

        with torch.no_grad():
            for images, labels, _paths in val_dl:
                images = images.to(DEVICE, non_blocking=True)
                logits = model(images)
                batch_correct, batch_total, pred_texts = exact_match_acc(logits, labels)
                exact_correct += batch_correct
                exact_total += batch_total

                batch_char_correct, batch_char_total = char_accuracy(pred_texts, labels)
                char_correct += batch_char_correct
                char_total += batch_char_total

                if sample_pred is None:
                    sample_pred = pred_texts[:5]
                    sample_gt = labels[:5]

        avg_loss = train_loss / max(1, len(train_dl))
        val_exact = exact_correct / max(1, exact_total)
        val_char = char_correct / max(1, char_total)

        scheduler.step(val_char)
        lr_string = ", ".join(f"{group['lr']:.6f}" for group in optimizer.param_groups)
        print(
            f"Epoch {epoch:3d}/{EPOCHS} | "
            f"loss={avg_loss:.4f} | "
            f"exact_acc={val_exact:.3f} ({exact_correct}/{exact_total}) | "
            f"char_acc={val_char:.3f} | "
            f"lr=[{lr_string}]"
        )

        if sample_pred is not None and epoch in {1, 2, 5, 10, 20, 40, 60, 80, 100, 120}:
            print("  GT :", sample_gt)
            print("  PR :", sample_pred)

        if should_replace_best(best_exact, best_char, val_exact, val_char):
            best_exact = val_exact
            best_char = val_char
            epochs_without_improvement = 0
            torch.save(model.state_dict(), SAVE_PATH)
            print(f"Saved best model: exact={best_exact:.3f}, char={best_char:.3f}")
        else:
            epochs_without_improvement += 1

        if epoch >= MIN_EPOCHS and epochs_without_improvement >= EARLY_STOP_PATIENCE:
            print(
                f"Early stopping at epoch {epoch}: "
                f"no validation improvement for {epochs_without_improvement} epochs."
            )
            break

    print(f"\nDone. Best exact acc: {best_exact:.3f}")
    print(f"Best char acc: {best_char:.3f}")
    print(f"Model saved to: {SAVE_PATH}")


if __name__ == "__main__":
    import torch.multiprocessing as mp

    mp.freeze_support()
    main()
